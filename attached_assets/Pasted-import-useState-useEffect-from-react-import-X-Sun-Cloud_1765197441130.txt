import { useState, useEffect } from 'react';
import { X, Sun, Cloud, CloudRain, CloudSnow, Wind, Droplets, Thermometer, TrendingUp, Heart, Moon, Activity, Zap, MessageCircle, ThumbsUp, ThumbsDown } from 'lucide-react';

interface MorningBriefingProps {
  isDark: boolean;
  onClose: () => void;
  userName?: string;
  onLaunchVoiceChat?: () => void;
}

interface WeatherData {
  temp: number;
  condition: string;
  description: string;
  humidity: number;
  feelsLike: number;
}

interface BriefingData {
  readinessScore: number;
  sleepData: {
    totalHours: number;
    deepSleepMinutes: number;
    deepSleepQuality: 'excellent' | 'good' | 'fair';
    hrvAvg: number;
  };
  recentActivity: {
    type: string;
    when: string;
    impact: string;
  };
  recommendation: string;
}

export function MorningBriefing({ isDark, onClose, userName = 'Alex', onLaunchVoiceChat }: MorningBriefingProps) {
  const [isVisible, setIsVisible] = useState(false);
  const [weather, setWeather] = useState<WeatherData | null>(null);
  const [isLoadingWeather, setIsLoadingWeather] = useState(true);
  const [feedbackGiven, setFeedbackGiven] = useState<'up' | 'down' | null>(null);

  // Mock briefing data - in production this would come from your backend/state
  const briefing: BriefingData = {
    readinessScore: 82,
    sleepData: {
      totalHours: 7.5,
      deepSleepMinutes: 98,
      deepSleepQuality: 'excellent',
      hrvAvg: 62
    },
    recentActivity: {
      type: 'Zone 2 run',
      when: 'yesterday',
      impact: 'positive recovery'
    },
    recommendation: "It's 45°F and rainy, so your planned outdoor run might be better on the treadmill. Consider a light Zone 2 session to maintain your cardiovascular fitness while giving your body the recovery it needs. Your HRV indicates you're well-recovered from yesterday's workout, so you have good capacity for moderate exercise today. Just remember to stay hydrated and fuel properly before your session."
  };

  // Fetch weather data
  useEffect(() => {
    // Mock weather data for now - replace with actual OpenWeather API call
    const fetchWeather = async () => {
      try {
        // TODO: Replace with actual API call
        // const apiKey = process.env.OPENWEATHER_API_KEY;
        // const response = await fetch(`https://api.openweathermap.org/data/2.5/weather?lat={lat}&lon={lon}&appid=${apiKey}&units=imperial`);
        // const data = await response.json();
        
        // Mock data
        setTimeout(() => {
          setWeather({
            temp: 45,
            condition: 'Rain',
            description: 'Light rain',
            humidity: 78,
            feelsLike: 42
          });
          setIsLoadingWeather(false);
        }, 800);
      } catch (error) {
        console.error('Failed to fetch weather:', error);
        setIsLoadingWeather(false);
      }
    };

    fetchWeather();
  }, []);

  // Entrance animation
  useEffect(() => {
    setTimeout(() => setIsVisible(true), 100);
  }, []);

  const getWeatherIcon = (condition: string) => {
    const lower = condition.toLowerCase();
    if (lower.includes('rain')) return CloudRain;
    if (lower.includes('cloud')) return Cloud;
    if (lower.includes('snow')) return CloudSnow;
    if (lower.includes('wind')) return Wind;
    return Sun;
  };

  const getReadinessColor = (score: number) => {
    if (score >= 80) return isDark ? 'text-green-400' : 'text-green-600';
    if (score >= 60) return isDark ? 'text-yellow-400' : 'text-yellow-600';
    return isDark ? 'text-orange-400' : 'text-orange-600';
  };

  const getReadinessGradient = (score: number) => {
    if (score >= 80) return 'from-green-500/20 to-emerald-500/20';
    if (score >= 60) return 'from-yellow-500/20 to-orange-500/20';
    return 'from-orange-500/20 to-red-500/20';
  };

  const handleClose = () => {
    setIsVisible(false);
    setTimeout(onClose, 300);
  };

  const handleFeedback = (type: 'up' | 'down') => {
    setFeedbackGiven(type);
    // TODO: Send feedback to backend/ML system
    console.log(`User feedback: ${type} for morning briefing`);
  };

  const handleVoiceChat = () => {
    if (onLaunchVoiceChat) {
      onLaunchVoiceChat();
    } else {
      // Fallback if no voice chat handler provided
      console.log('Voice chat launched for morning briefing');
    }
  };

  const WeatherIcon = weather ? getWeatherIcon(weather.condition) : Cloud;
  const currentHour = new Date().getHours();
  const greeting = currentHour < 12 ? 'Good morning' : currentHour < 18 ? 'Good afternoon' : 'Good evening';

  return (
    <div className="fixed inset-0 z-[200] flex items-center justify-center p-4">
      {/* Backdrop */}
      <div 
        className={`absolute inset-0 bg-black/60 backdrop-blur-md transition-opacity duration-300 ${
          isVisible ? 'opacity-100' : 'opacity-0'
        }`}
        onClick={handleClose}
      />
      
      {/* Modal - with overflow scroll */}
      <div 
        className={`relative w-full max-w-md max-h-[90vh] overflow-y-auto rounded-3xl shadow-2xl transition-all duration-500 ${
          isVisible ? 'scale-100 opacity-100 translate-y-0' : 'scale-95 opacity-0 translate-y-4'
        } ${
          isDark 
            ? 'bg-gradient-to-br from-slate-900/95 to-slate-800/95' 
            : 'bg-gradient-to-br from-white/95 to-gray-50/95'
        } backdrop-blur-2xl border ${
          isDark ? 'border-white/10' : 'border-black/10'
        }`}
      >
        {/* Close button */}
        <button
          onClick={handleClose}
          className={`absolute top-4 right-4 z-10 p-2 rounded-xl transition-colors ${
            isDark ? 'hover:bg-white/10 text-white/70' : 'hover:bg-black/5 text-gray-600'
          }`}
        >
          <X className="w-5 h-5" />
        </button>

        {/* Content */}
        <div className="p-6 space-y-5">
          {/* Header - Greeting */}
          <div className="text-center pt-2">
            <div className={`inline-flex items-center justify-center w-16 h-16 rounded-2xl mb-3 ${
              isDark 
                ? 'bg-gradient-to-br from-purple-500/20 to-pink-500/20' 
                : 'bg-gradient-to-br from-purple-100 to-pink-100'
            }`}>
              <Sun className={`w-8 h-8 ${isDark ? 'text-yellow-400' : 'text-yellow-500'}`} />
            </div>
            <h2 className={`text-xl mb-1 ${isDark ? 'text-white' : 'text-gray-900'}`}>
              {greeting}, {userName}
            </h2>
            <p className={`text-sm ${isDark ? 'text-white/60' : 'text-gray-600'}`}>
              Here's your morning briefing
            </p>
          </div>

          {/* Readiness Score - Hero */}
          <div className={`rounded-2xl border p-5 bg-gradient-to-br ${getReadinessGradient(briefing.readinessScore)} ${
            isDark ? 'border-white/10' : 'border-black/5'
          }`}>
            <div className="flex items-center justify-between mb-3">
              <div className="flex items-center gap-2">
                <Zap className={`w-5 h-5 ${getReadinessColor(briefing.readinessScore)}`} />
                <span className={`text-sm ${isDark ? 'text-white/80' : 'text-gray-700'}`}>
                  Readiness Score
                </span>
              </div>
              <div className={`text-3xl ${getReadinessColor(briefing.readinessScore)}`}>
                {briefing.readinessScore}
              </div>
            </div>
            <div className={`h-2 rounded-full overflow-hidden ${
              isDark ? 'bg-white/10' : 'bg-black/10'
            }`}>
              <div 
                className={`h-full transition-all duration-1000 ${
                  briefing.readinessScore >= 80 ? 'bg-green-500' :
                  briefing.readinessScore >= 60 ? 'bg-yellow-500' :
                  'bg-orange-500'
                }`}
                style={{ 
                  width: `${briefing.readinessScore}%`,
                  transitionDelay: '300ms'
                }}
              />
            </div>
          </div>

          {/* Sleep Summary */}
          <div className={`rounded-2xl border p-4 ${
            isDark ? 'bg-white/5 border-white/10' : 'bg-gray-50 border-gray-200'
          }`}>
            <div className="flex items-start gap-3">
              <div className={`p-2 rounded-xl ${
                isDark ? 'bg-purple-500/20' : 'bg-purple-100'
              }`}>
                <Moon className={`w-5 h-5 ${isDark ? 'text-purple-400' : 'text-purple-600'}`} />
              </div>
              <div className="flex-1">
                <div className={`text-sm mb-1 ${isDark ? 'text-white' : 'text-gray-900'}`}>
                  Sleep Quality
                </div>
                <p className={`text-xs leading-relaxed ${isDark ? 'text-white/70' : 'text-gray-600'}`}>
                  Your deep sleep was <span className={isDark ? 'text-purple-400' : 'text-purple-600'}>
                    {briefing.sleepData.deepSleepQuality}
                  </span> ({briefing.sleepData.deepSleepMinutes} min), likely driven by that {briefing.recentActivity.type} {briefing.recentActivity.when}.
                </p>
                <div className="flex items-center gap-4 mt-2">
                  <div className={`text-xs ${isDark ? 'text-white/50' : 'text-gray-500'}`}>
                    <span className={isDark ? 'text-white/80' : 'text-gray-700'}>{briefing.sleepData.totalHours}h</span> total
                  </div>
                  <div className={`text-xs ${isDark ? 'text-white/50' : 'text-gray-500'}`}>
                    <span className={isDark ? 'text-white/80' : 'text-gray-700'}>{briefing.sleepData.hrvAvg}</span> HRV avg
                  </div>
                </div>
              </div>
            </div>
          </div>

          {/* Weather */}
          {!isLoadingWeather && weather && (
            <div className={`rounded-2xl border p-4 ${
              isDark 
                ? 'bg-gradient-to-br from-cyan-500/10 to-blue-500/10 border-cyan-500/30' 
                : 'bg-gradient-to-br from-cyan-50 to-blue-50 border-cyan-200'
            }`}>
              <div className="flex items-start gap-3">
                <div className={`p-2 rounded-xl ${
                  isDark ? 'bg-cyan-500/20' : 'bg-cyan-100'
                }`}>
                  <WeatherIcon className={`w-5 h-5 ${isDark ? 'text-cyan-400' : 'text-cyan-600'}`} />
                </div>
                <div className="flex-1">
                  <div className="flex items-baseline gap-2 mb-1">
                    <span className={`text-2xl ${isDark ? 'text-white' : 'text-gray-900'}`}>
                      {Math.round(weather.temp)}°F
                    </span>
                    <span className={`text-sm ${isDark ? 'text-white/60' : 'text-gray-600'}`}>
                      {weather.description}
                    </span>
                  </div>
                  <div className="flex items-center gap-3">
                    <div className={`flex items-center gap-1 text-xs ${isDark ? 'text-white/50' : 'text-gray-500'}`}>
                      <Thermometer className="w-3.5 h-3.5" />
                      Feels {Math.round(weather.feelsLike)}°
                    </div>
                    <div className={`flex items-center gap-1 text-xs ${isDark ? 'text-white/50' : 'text-gray-500'}`}>
                      <Droplets className="w-3.5 h-3.5" />
                      {weather.humidity}%
                    </div>
                  </div>
                </div>
              </div>
            </div>
          )}

          {/* Recommendation - dynamic height */}
          <div className={`rounded-2xl border p-4 ${
            isDark 
              ? 'bg-gradient-to-br from-orange-500/10 to-pink-500/10 border-orange-500/30' 
              : 'bg-gradient-to-br from-orange-50 to-pink-50 border-orange-200'
          }`}>
            <div className="flex items-start gap-3">
              <div className={`p-2 rounded-xl flex-shrink-0 ${
                isDark ? 'bg-orange-500/20' : 'bg-orange-100'
              }`}>
                <Activity className={`w-5 h-5 ${isDark ? 'text-orange-400' : 'text-orange-600'}`} />
              </div>
              <div className="flex-1 min-w-0">
                <div className={`text-sm mb-1 ${isDark ? 'text-white' : 'text-gray-900'}`}>
                  Today's Recommendation
                </div>
                <p className={`text-xs leading-relaxed ${isDark ? 'text-white/70' : 'text-gray-600'}`}>
                  {briefing.recommendation}
                </p>
              </div>
            </div>
          </div>

          {/* Feedback Buttons */}
          <div className="flex items-center justify-center gap-4">
            <button
              onClick={() => handleFeedback('up')}
              className={`p-3 rounded-xl transition-all ${
                feedbackGiven === 'up' 
                  ? 'bg-green-500 text-white shadow-lg scale-110' 
                  : isDark
                    ? 'bg-white/10 text-white/60 hover:bg-white/20'
                    : 'bg-gray-200 text-gray-500 hover:bg-gray-300'
              }`}
            >
              <ThumbsUp className="w-5 h-5" />
            </button>
            <button
              onClick={() => handleFeedback('down')}
              className={`p-3 rounded-xl transition-all ${
                feedbackGiven === 'down' 
                  ? 'bg-red-500 text-white shadow-lg scale-110' 
                  : isDark
                    ? 'bg-white/10 text-white/60 hover:bg-white/20'
                    : 'bg-gray-200 text-gray-500 hover:bg-gray-300'
              }`}
            >
              <ThumbsDown className="w-5 h-5" />
            </button>
          </div>

          {/* Voice Chat Button - with pink gradient */}
          <button
            onClick={handleVoiceChat}
            className={`w-full py-3.5 rounded-2xl transition-all flex items-center justify-center gap-2 ${
              isDark 
                ? 'bg-gradient-to-r from-purple-500 to-pink-500 hover:from-purple-600 hover:to-pink-600 text-white' 
                : 'bg-gradient-to-r from-purple-600 to-pink-600 hover:from-purple-700 hover:to-pink-700 text-white'
            } shadow-lg hover:shadow-xl`}
          >
            <MessageCircle className="w-5 h-5" />
            <span>Get your briefing from Flō</span>
          </button>
        </div>
      </div>
    </div>
  );
}
